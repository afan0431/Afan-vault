-- ========== 1. 常量与配置定义 ==========
local SETTINGS = {
    Keys = {"AfanSAM", "Team1", "Team2", "爆发药", "疾跑", "真北"},
    Save_Path = GetLuaModsPath() .. [[/AfanSAM_Settings.txt]], 
    Skills = {
        IKISHOTEN = 16482,    -- 意气冲天
        SHOHA = 16487,        -- 照破
        YUKIKAZE = 7480,      -- 雪风
        NAMIKIRI = 25781,     -- 奥义斩浪
        Gyofu = 36963,        -- 晓风
        TENDO_KAESHI = 36986  -- 天道回返雪月花
    },
    Buffs = {
        TSUBAME = 4216,       -- 回返预备
        HIGANBANA = 1228,     -- 彼岸花
        OGI_READY = 2959,     -- 奥义斩浪预备 (意气开启后的Buff)
    }
}

-- ========== 2. 数据初始化与持久化 ==========

local function saveAfanState(state)
    local file = io.open(SETTINGS.Save_Path, "w")
    if not file then return end
    for btnName, config in pairs(state) do
        if type(config) == "table" and config.bool ~= nil then
            file:write(string.format("%s=%s\n", btnName, tostring(config.bool)))
        end
    end
    file:close()
end

local function loadAfanState()
    local state = {}
    local file = io.open(SETTINGS.Save_Path, "r")
    if not file then return state end
    for line in file:lines() do
        local btnName, boolStr = string.match(line, "^([^=]+)=(%a+)$")
        if btnName and boolStr then
            local boolVal = (boolStr == "true")
            state[btnName] = {bool = boolVal}
        end
    end
    file:close()
    return state
end

if not data then data = {} end
if data.AfanSAM == nil then
    data.AfanSAM = loadAfanState()
end

-- 补全默认按钮
local defaultBtns = {
    ["AfanSAM"] = true,
    ["Team1"] = true,
    ["Team2"] = false,
    ["爆发药"] = true,
    ["疾跑"] = true,
    ["真北"] = true
}

local needInitialSave = false
for btnName, defaultBool in pairs(defaultBtns) do
    if not data.AfanSAM[btnName] then
        data.AfanSAM[btnName] = {bool = defaultBool}
        needInitialSave = true
    end
end
if needInitialSave then saveAfanState(data.AfanSAM) end

-- ========== 3. 核心战斗逻辑 ==========
local function handleAfanCombatLogic()
    local player = TensorCore.mGetPlayer()
    if not player or player.job ~= 34 then return end
    
    -- 开关未开启则重置
    if not data.AfanSAM["AfanSAM"].bool then
        data.shohaHoldEndTime = 0
        data.waitingForTendoKaeshi = false
        _G["ACR_TensorWeeb3_HoldTsubame"] = false
        return 
    end

    local target = TensorCore.mGetTarget()
    local now = Now()
    local combatTimer = TensorReactions_CurrentCombatTimer or 0

    -- 技能数据
    local ikishoten = ActionList:Get(1, SETTINGS.Skills.IKISHOTEN)
    local shoha = ActionList:Get(1, SETTINGS.Skills.SHOHA)
    local ikishotenCd = ikishoten and (ikishoten.cdmax - ikishoten.cd) or 0

    -- A. 优先雪闪逻辑
    if player.lastcomboid == SETTINGS.Skills.Gyofu and player.gauge[2] <= 0 and target then
        ActionList:Get(1, SETTINGS.Skills.YUKIKAZE):Cast(target.id)
    end

    -- B. 照破资源管理
    local hasThreeSen = player.gauge[2] >= 7
    local hasTsubameBuff = TensorCore.getBuff(player, SETTINGS.Buffs.TSUBAME) ~= nil
    local nextIsSetsugekka = hasThreeSen and not hasTsubameBuff

    if shoha and shoha.highlighted > 0 and (ikishotenCd <= 30 or ikishotenCd >= 110) and not nextIsSetsugekka then
        data.shohaHoldEndTime = now + 2500
        TensorCore.API.TensorACR.holdActionUntil(SETTINGS.Skills.SHOHA, data.shohaHoldEndTime)
    else
        data.shohaHoldEndTime = now
        TensorCore.API.TensorACR.holdActionUntil(SETTINGS.Skills.SHOHA, now)
    end

	-- C. 奥义斩浪 (Namikiri) 精准对齐逻辑
    if data.waitingForTendoKaeshi == nil then data.waitingForTendoKaeshi = false end
    
    local hasOgiBuff = TensorCore.getBuff(player, SETTINGS.Buffs.OGI_READY) ~= nil
    local hasTsubameBuff = TensorCore.getBuff(player, SETTINGS.Buffs.TSUBAME) ~= nil
    local lastID = player.lastcastid
    local lastUsed = ActionList:LastUsedID() -- 增加一种 ID 获取方式

    -- 1. 开启锁定：当同时拥有“斩浪预备”和“回返预备”时，必须进入等待状态
    if hasOgiBuff and hasTsubameBuff then
        data.waitingForTendoKaeshi = true
    end

    -- 2. 解除锁定（多重保险）：
    -- 条件 A：上一个技能是天道回返 (36986)
    -- 条件 B：回返 Buff 已经消失（说明已经打出或过期）
    -- 条件 C：斩浪 Buff 已经消失（说明已经打出或过期）
    if lastID == SETTINGS.Skills.TENDO_KAESHI 
       or lastUsed == SETTINGS.Skills.TENDO_KAESHI 
       or not hasTsubameBuff 
       or not hasOgiBuff 
    then
        data.waitingForTendoKaeshi = false
    end

    -- 3. 执行动作持有
    if data.waitingForTendoKaeshi then
        -- 将延迟时间缩短为 500ms，每帧刷新。
        -- 这样即使逻辑切换到 false，拦截也会在 0.5 秒内立即消失，不会卡死技能。
        TensorCore.API.TensorACR.holdActionUntil(SETTINGS.Skills.NAMIKIRI, now + 500)
    end

    -- D. 回返 (Tsubame) 延后控制逻辑
    if combatTimer > 60 then
        -- 获取目标彼岸花 Dot 剩余时间
        local higanbana = target and TensorCore.getBuff(target, SETTINGS.Buffs.HIGANBANA)
        local dotTime = higanbana and higanbana.duration or 0

        if ikishotenCd > 119 then
            -- 意气刚开启，允许释放回返
            _G["ACR_TensorWeeb3_HoldTsubame"] = false

        elseif combatTimer <= 360 and dotTime < 13 then
            -- 仅在 360s 内执行：若彼岸花时间紧迫，释放回返以对齐循环
            _G["ACR_TensorWeeb3_HoldTsubame"] = false

        elseif ikishotenCd < 90 then
            -- 爆发准备期：延后回返
            _G["ACR_TensorWeeb3_HoldTsubame"] = true

        else
            -- 默认情况及 360s 后
            _G["ACR_TensorWeeb3_HoldTsubame"] = false
        end
    else
        -- 战斗时间不足 60s
        _G["ACR_TensorWeeb3_HoldTsubame"] = false
    end 
end

-- ========== 4. GUI 渲染模块 ==========
local function drawAfanGUI()
    local flags = GUI.WindowFlags_NoTitleBar + GUI.WindowFlags_NoScrollbar + GUI.WindowFlags_AlwaysAutoResize
    
    -- 窗口1：总控
    GUI:Begin("AfanSAM_Main", true, flags)
    local mainInfo = data.AfanSAM["AfanSAM"]
    local mainColor = mainInfo.bool and {r=0, g=1, b=0, a=0.5} or {r=1, g=0.07, b=0, a=0.5}
    
    GUI:PushStyleVar(GUI.StyleVar_ChildWindowRounding, 5)
    GUI:PushStyleColor(GUI.Col_ChildWindowBg, mainColor.r, mainColor.g, mainColor.b, mainColor.a)
    
    GUI:BeginChild("AfanSAM_Btn", 100, 30, true)
        local tw, th = GUI:CalcTextSize("AfanSAM")
        GUI:SetCursorPosX((100 - tw) * 0.5)
        GUI:SetCursorPosY((30 - th) * 0.5)
        GUI:Text("AfanSAM")
    GUI:EndChild()
    
    if GUI:IsItemHovered() and GUI:IsMouseClicked(0) then
        mainInfo.bool = not mainInfo.bool
        saveAfanState(data.AfanSAM)
    end
    
    GUI:PopStyleColor()
    GUI:Separator()
    
    if data.shohaHoldEndTime and data.shohaHoldEndTime > Now() then
        GUI:TextColored(1, 0.8, 0.2, 1, "照破延迟到下一个雪月花")
    elseif data.waitingForTendoKaeshi then
        GUI:TextColored(0.4, 0.8, 1, 1, "等待天道回返中...")
    else
        GUI:TextDisabled("技能无延迟")
    end
    GUI:End()

    -- 窗口2：功能开关
    GUI:Begin("AfanSAM_Toggles", true, flags)
    local subButtons = {"Team1", "Team2", "爆发药", "疾跑", "真北"}
    
    for _, name in ipairs(subButtons) do
        local info = data.AfanSAM[name]
        local color = info.bool and {r=0, g=1, b=0, a=0.5} or {r=1, g=0.07, b=0, a=0.5}
        
        GUI:PushStyleColor(GUI.Col_ChildWindowBg, color.r, color.g, color.b, color.a)
        GUI:BeginChild(name, 100, 30, true)
            local btw, bth = GUI:CalcTextSize(name)
            GUI:SetCursorPosX((100 - btw) * 0.5)
            GUI:SetCursorPosY((30 - bth) * 0.5)
            GUI:Text(name)
        GUI:EndChild()
        
        if GUI:IsItemHovered() and GUI:IsMouseClicked(0) then
            info.bool = not info.bool
            if info.bool then
                if name == "Team1" then data.AfanSAM["Team2"].bool = false 
                elseif name == "Team2" then data.AfanSAM["Team1"].bool = false end
            end
            saveAfanState(data.AfanSAM)
        end
        GUI:PopStyleColor()
    end
    
    GUI:PopStyleVar(1)
    GUI:End()
end

-- ========== 5. 主流程执行 ==========
drawAfanGUI()
handleAfanCombatLogic()

self.used = true
