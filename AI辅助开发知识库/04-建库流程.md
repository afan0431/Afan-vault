# 建库流程

建库是一个离线的批处理过程：读取知识文档，生成向量，写入 Qdrant。通常只在知识库初始化或知识内容变更时触发，不是实时操作。

---

## 整体流程

```
knowledge/ 目录下的 .md 文件
        ↓
解析 Frontmatter 元数据
        ↓
提取正文内容，拼接"标题 + 正文"作为 Embedding 输入
        ↓
调用 Embedding API 生成向量
        ↓
Upsert 到 Qdrant（已存在则覆盖，新增则插入）
```

---

## 知识文件格式规范

建库脚本依赖 Frontmatter 提取元数据，每个知识文件**必须**包含标准格式的 Frontmatter：

```markdown
---
category: architecture
domain: order
priority: high
tags: [cqrs, mediatr]
last_verified: 2026-03-01
---

# 为什么采用 CQRS 模式

正文内容从这里开始...
```

没有 Frontmatter 的文件会被建库脚本跳过。

## 建库脚本

```python
# index_knowledge.py
import os
import yaml
import httpx
import hashlib
from pathlib import Path
from datetime import datetime

# ===== 配置 =====
KNOWLEDGE_DIR = "./knowledge"
QDRANT_URL = "http://localhost:6333"
COLLECTION = "dev_knowledge"

# 选择 Embedding 方案：openai 或 ollama
EMBEDDING_PROVIDER = os.environ.get("EMBEDDING_PROVIDER", "openai")
OPENAI_API_KEY = os.environ.get("OPENAI_API_KEY", "")
OPENAI_MODEL = "text-embedding-3-small"
OLLAMA_URL = os.environ.get("OLLAMA_URL", "http://localhost:11434")
OLLAMA_MODEL = "bge-m3"

# ===== Embedding =====

def embed_openai(text: str) -> list[float]:
    resp = httpx.post(
        "https://api.openai.com/v1/embeddings",
        headers={"Authorization": f"Bearer {OPENAI_API_KEY}"},
        json={"input": text, "model": OPENAI_MODEL},
        timeout=30,
    )
    resp.raise_for_status()
    return resp.json()["data"][0]["embedding"]


def embed_ollama(text: str) -> list[float]:
    resp = httpx.post(
        f"{OLLAMA_URL}/api/embeddings",
        json={"model": OLLAMA_MODEL, "prompt": text},
        timeout=60,
    )
    resp.raise_for_status()
    return resp.json()["embedding"]


def get_embedding(text: str) -> list[float]:
    if EMBEDDING_PROVIDER == "ollama":
        return embed_ollama(text)
    return embed_openai(text)


# ===== 文档解析 =====

def parse_md(path: Path) -> dict | None:
    content = path.read_text(encoding="utf-8")

    if not content.startswith("---"):
        return None

    end = content.find("---", 3)
    if end == -1:
        return None

    try:
        frontmatter = yaml.safe_load(content[3:end])
    except yaml.YAMLError:
        return None

    body = content[end + 3:].strip()

    # 提取第一个 H1 作为标题
    title = next(
        (line[2:].strip() for line in body.splitlines() if line.startswith("# ")),
        path.stem,
    )

    return {
        "title": title,
        "content": body,
        "embed_text": f"{title}\n\n{body}",
        "metadata": {
            "source": str(path.relative_to(KNOWLEDGE_DIR)),
            "category": frontmatter.get("category", ""),
            "domain": frontmatter.get("domain", ""),
            "priority": frontmatter.get("priority", "medium"),
            "tags": frontmatter.get("tags", []),
            "last_verified": str(frontmatter.get("last_verified", "")),
        },
    }


# ===== Qdrant 写入 =====

def stable_id(path: Path) -> str:
    """基于文件路径生成稳定的数字 ID（Qdrant 要求数字或 UUID）"""
    return str(int(hashlib.md5(str(path).encode()).hexdigest(), 16) % (10**15))


def upsert(doc_id: str, vector: list[float], payload: dict):
    httpx.put(
        f"{QDRANT_URL}/collections/{COLLECTION}/points",
        json={"points": [{"id": doc_id, "vector": vector, "payload": payload}]},
        timeout=15,
    ).raise_for_status()


# ===== 主流程 =====

def index_all():
    files = list(Path(KNOWLEDGE_DIR).rglob("*.md"))
    print(f"发现 {len(files)} 个文件")

    success = skipped = 0
    for i, path in enumerate(files, 1):
        doc = parse_md(path)
        if not doc:
            print(f"  [{i}/{len(files)}] 跳过（无 Frontmatter）：{path.name}")
            skipped += 1
            continue

        vector = get_embedding(doc["embed_text"])
        payload = {
            **doc["metadata"],
            "title": doc["title"],
            "content": doc["content"],
            "indexed_at": datetime.now().isoformat(),
        }
        upsert(stable_id(path), vector, payload)
        print(f"  [{i}/{len(files)}] ✓ {doc['title']}")
        success += 1

    print(f"\n完成：入库 {success} 条，跳过 {skipped} 条")


if __name__ == "__main__":
    index_all()
```

## 增量更新

全量重建对于小规模知识库（< 1000 条）通常只需几十秒，是最简单的做法。

如果知识库规模较大或 Embedding API 调用成本敏感，可以做增量更新——只处理变更的文件：

```bash
# 找出相对上次提交有变更的知识文件
git diff --name-only HEAD | grep "^knowledge/.*\.md$"
```

配合建库脚本的 `--files` 参数（可自行扩展），只对这些文件重新入库。
